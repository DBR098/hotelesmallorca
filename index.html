function muestraOculta(id) { var elemento = document.getElementById('contenidos_'+id); var enlace = document.getElementById('enlace_'+id); if(elemento.style.display == "" || elemento.style.display == "block") { elemento.style.display = "none"; enlace.innerHTML = 'Mostrar contenidos'; } else { elemento.style.display = "block"; enlace.innerHTML = 'Ocultar contenidos'; } }

Hotel Diamante

Diamante es un pequeño hotel familiar céntrico a tan solo 100 m. de la Playa Palmira en una tranquila calle adyacente al Bulevar en donde se encuentran numerosas posibilidades de entretenimiento como pasear, ir de compras o comer y beber algo sentado en una de las terrazas de los numerosos bares y restaurantes.
